Enables autocomplete on the field element

var view = new Maya.Views.InviteForm({ model_class: 'EventJudge' })
view.enableAutComplete('.input')

  enableAutoComplete: (field) ->
    $(field).autocomplete
      source: "/users"
      select: (event, ui) ->
        $(this).val(ui.item.full_name)
        $(this).data("guest_id", ui.item.id)
        return false
      response: (event, ui) ->
        unless ui.content.length > 0
          $(this).trigger("noresults", [ $(this).val() ])
        else
          $(this).trigger("results", [ ui.content ])
    .data( "autocomplete" )
    ._renderItem = ( ul, item ) ->
      $( "<li>" )
        .data( "item.autocomplete", item )
        .append( JST['invites/user_item'](item) )
        .appendTo( ul )

Displays the email field with animtion

 
  displayEmail: (event, fullname, speed = 50) ->
    @$el.find('.user-fullname')
      .animate( width: @input_width, speed)
    @$el.find('.user-email')
      .css(visibility: 'visible')
      .animate(width: @input_width, speed)
      .attr('placeholder', "#{fullname}'s email")
    @$el.find('#submit-button').text("Invite")

  hideEmail: ->
    @$el.find('.user-fullname')
      .animate( width: @input_width * 2, 100)
    @$el.find('.user-email')
      .animate(width: 1, 100)
      .css(visibility: 'hidden')
    @$el.find('#submit-button').text("Add")

  triggerSuccess: ->
    this.trigger('user:added')

  submitClicked: (event) ->
    event.preventDefault()
    @invite = new Models.Invitation
      purpose: @data.purpose
      purpose_data_id: @data.purpose_data_id
      email:     this.form().email
      full_name: this.form().full_name
      guest_id:  this.form().guest_id
    @invite.on 'change', this.triggerSuccess, this
    @invite.on 'sync',  this.resetForm, this
    @invite.save()

  form: ->
    attrs = {}
    for field in @$el.find('input[name]')
      attrs[$(field).attr('name')] = $(field).val()
    attrs.guest_id = $('#user-fullname').data("guest_id")
    attrs

  resetForm: ->
    for field in @$el.find('input[name]')
      $(field).val("")
    $('#user-fullname').data("guest_id", "")
